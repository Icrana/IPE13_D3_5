// file gateway.rules
rule "alarmByLightBar_rule"
	when 
		Item Genuino_lightbar from OFF to 0N
	then	
		alarmByLightBar.postUpdate("ACTIVATED")
end

	when 
		Item Genuino_button from ON to OFF
	then
		if (alarmByLightBar == ON){
			alarmByLightBar.postUpdate("deactivated"
		} 
		else {
		}
end
